@startuml

+interface IGameObject {
    name() : String
    transform() : ITransform
    collider() : ICollider
}

+interface ITransform {
    +move() : void
    +rotate() : void
    +scale() : void

    +position() : Point
    +layer() : int
    +angle() : double
    +scale() : double
}

+interface ICollider {
    centroid() : Point
}

+class GameEngine {
    +add() : void
    +destroy() : void

    +objects() : ArrayList<GameObject>
}

+class GameObject implements IGameObject{
    +name : String
    +transform : Transform
    +collider : Collider

    +move() : void
    +rotate() : void
    +scale() : void
    +checkCollision() : boolean

    +toString() : String
    +equals() : boolean
    +hashCode() : boolean
}

+class Transform implements ITransform {
    -layer : int
    -angle : double
    -scale : double

    +toString() : String
}

+abstract class Collider implements ICollider {
    #scale : double
    #angle : double

    +{abstract} move() : void
    +{abstract} rotate() : void
    +{abstract} scale() : void
    +{abstract} checkCollision() : boolean
    +checkCollisionCircle() : boolean
    +checkCollisionPolygon() : boolean

    +toString() : String
}

+class Point {
    -x : double
    -y : double

    +addNew() : Point
    +subNew() : Point
    +multNew() : Point
    +addThis() : void
    +subThis() : void
    +multThis() : void
    +rotateThis() : void
    +scaleThis() : void
    +distFrom() : double

    +x() : double
    +y() : double

    +toString() : String
    +equals() : boolean
    +clone() : Point
}

+class ColliderCircle extends Collider {
    -radius : double

    -{static} checkRadius() : void
    +segmentIntersect() : boolean

    +radius() : double
}

+class ColliderPolygon extends Collider {
    -{static} checkVertices() : void
    -calculateArea() : double
    -calculateCentroid() : Point
    -checkPointInside() : boolean

    +vertices() : ArrayList<Point>
    +edges() : ArrayList<LineSegment>
}

+class Line{
    -{static} checkEquals() : void
    #pointOrientation() : orientation
    +pointCollinear : boolean
    +closestPointFromPoint : Point
}

+class LineSegment extends Line{
    +pointOnSegment() : boolean
    +segmentIntersect() : boolean

    +p1() : Point
    +p2() : Point

    +toString() : String
}

GameEngine o-- GameObject : -ArrayList<> objects
GameObject o-- Transform : -transform
GameObject o-- Collider : -collider
Transform *-- Point : -position
Collider *-- Point : #centroid
ColliderPolygon *-- Point : -ArrayList<> vertices
ColliderPolygon *-- LineSegment : -ArrayList<> edges
Line *-- Point : #p1, p2

@enduml
